CREATE VIEW view_candidatosp as SELECT cu.idcandidato, cu.fecha, CONCAT(cr.nombre, ' ', cr.apaterno, ' ', cr.amaterno) as candidato, CONCAT(cr.calle, ' ', cr.numext, ' ', cr.numint, ' ', cr.colonia, ' ', cr.ciudad, ' ', ce.estado) as direccion, cr.rfc, cr.telefono, cu.usuario as correo, cu.activo FROM candidato_usuarios as cu JOIN candidato_registro as cr on cu.idcandidato = cr.idcandidato JOIN catalogo_estados as ce ON cr.estado = ce.idestado
